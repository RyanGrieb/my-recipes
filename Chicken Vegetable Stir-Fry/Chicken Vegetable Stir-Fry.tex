\makeRecipeTitle{Chicken \& Vegetable Stir-Fry}

\section*{Ingredients}

\textbf{The Chicken (Marinade)}
\begin{itemize}[itemsep=0.1em, label={\raisebox{-0.1ex}{\large\textbullet}}]
    \item Chicken Breast or Thighs – 450g (1 lb), cut into bite-sized pieces
    \item Soy Sauce – 1/2 tbsp
    \item Shaoxing Wine – 1 tbsp (or dry sherry/cooking wine)
    \item White Pepper – 1/2 tsp
    \item Cornstarch – 1 tbsp
    \item Baking Soda – 1/4 tsp (tenderizer)
    \item Avocado Oil – 1 tbsp
\end{itemize}

\textbf{Vegetables \& Aromatics}
\begin{itemize}[itemsep=0.1em, label={\raisebox{-0.1ex}{\large\textbullet}}]
    \item Bell Peppers – 2 to 3, cut into bite-sized squares
    \item Green Onions – 2 to 3 stalks
    \item Garlic – 3 cloves, minced (or 1-2 tbsp jarred garlic)
    \item Ginger – 1 tsp, grated (fresh or powder)
    \item Optional Veggies – Onion, broccoli, or asparagus (bite-sized)
    \item Avocado Oil – 2 tbsp
\end{itemize}

\textbf{The Sauce}
\begin{itemize}[itemsep=0.1em, label={\raisebox{-0.1ex}{\large\textbullet}}]
    \item Soy Sauce – 1 tbsp (+ 1/2 tsp of Dark Soy Sauce, optional)
    \item Oyster Sauce – 1 tbsp
    \item Honey – 1/2 tbsp
    \item Shaoxing Wine – 1 tbsp
    \item Chicken Bouillon Powder – 1 tbsp (reduce if too salty)
    \item Water – 1/4 cup
    \item Cornstarch – 1 tsp
\end{itemize}

\textbf{The Finish}
\begin{itemize}[itemsep=0.1em, label={\raisebox{-0.1ex}{\large\textbullet}}]
    \item Sesame Oil – 2 tsp
\end{itemize}

\newpage
\makeRecipeTitle{Chicken \& Vegetable Stir-Fry}
\section*{Instructions}

\begin{enumerate}

    \item \textbf{Marinate the Chicken (10 minutes)} \\
          In a bowl, combine the chicken with the marinade ingredients (soy sauce,
          wine, white pepper, cornstarch, baking soda, and oil). Mix by hand until
          the liquid is absorbed. Place in the fridge for at least 10 minutes.

    \item \textbf{Prep Vegetables \& Sauce} \\
          \textbf{Vegetables:} Separate the white and green parts of the green onions.
          Thinly slice the whites; cut greens into 2-inch pieces. Chop bell peppers
          and mince the garlic and ginger. \\
          \textbf{Sauce:} In a small container, whisk together all sauce ingredients
          (soy sauces, oyster sauce, honey, wine, bouillon, water, and cornstarch)
          until smooth.

    \item \textbf{Sear the Chicken (5 minutes)} \\
          Heat 2 tablespoons of oil in a pan/wok on High heat. Ensure the pan is hot
          before adding chicken. Spread chicken evenly in a single layer. Sear
          undisturbed for 2 minutes until golden brown. Flip and stir-fry for 3
          minutes until cooked through. Remove chicken and set aside.

    \item \textbf{Aromatics \& Veggies (2--3 minutes)} \\
          Add 1 tablespoon of oil to the pan on Medium heat. Add the white parts of
          the green onion, garlic, and ginger. Mix for 1 minute until fragrant. \\
          Increase heat to High. Add bell peppers (and any optional veggies) and stir
          for 1 minute until fully coated and slightly tender.

    \item \textbf{Combine \& Thicken} \\
          Add the cooked chicken back to the pan and stir-fry for 30 seconds.
          Give the sauce a quick mix (cornstarch settles), then pour it over the
          ingredients. Stir for 1 minute or until the sauce bubbles and thickens
          into a glaze.

    \item \textbf{The Finish} \\
          Turn off the heat. Add the green parts of the green onion and 2 teaspoons
          of sesame oil. Mix well and serve immediately over rice.

\end{enumerate}
