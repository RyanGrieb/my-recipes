\section*{Ingredients}

\begin{itemize}[itemsep=0.2em, label={\raisebox{-0.1ex}{\large\textbullet}}]
	\item Sirloin Steak – 2 lbs (room temp), cubed
	\item Green Beans – 10–12 oz (steam-in-bag style)
	\item Avocado Oil – 3 tbsp (divided)
	\item Unsalted Butter – ½ cup (1 stick)
	\item Worcestershire Sauce – 2 tbsp
	\item Shallot – 1 small, minced
	\item Garlic Cloves – 6, minced
	\item Fresh Parsley – 1 tbsp, minced
	\item Fresh Rosemary – 2 tsp, minced
	\item Fresh Thyme – 2 tsp, minced
	\item Kosher Salt – 2 tsp (plus more for finishing)
	\item Black Pepper – 2 tsp (plus more for finishing)
	\item Garlic Powder – 2 tsp
\end{itemize}

\newpage
\makeRecipeTitle
\section*{Instructions}

\begin{enumerate}[label=\textbf{\arabic*.}, itemsep=0.25em]

	\item \textbf{Prep Green Beans} \\
	      Microwave the bag of green beans for about 2 minutes less than the package instructions. Drain and set aside.

	\item \textbf{Prep Steak} \\
	      Pat steak dry. Slice into 1-inch cubes. Toss with 1 tbsp avocado oil, kosher salt, black pepper, and garlic powder.

	\item \textbf{The Sear} \\
	      Heat skillet to High. Add 2 tbsp oil. Sear steak bites in batches (don't overcrowd) for 1 minute per side. Remove and set aside.

	\item \textbf{Sauce \& Veggies} \\
	      Reduce heat to Medium. Add butter. Sauté shallots (1 min). Add garlic, rosemary, and thyme (30 sec). Add green beans and toss to coat (1 min).

	\item \textbf{The Finish} \\
	      Return steak and juices to pan. Add Worcestershire sauce. Toss everything for 30–45 seconds. Garnish with parsley and serve immediately.

\end{enumerate}
