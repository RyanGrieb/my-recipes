\begin{itemize}[itemsep=0.2em, label={\raisebox{-0.1ex}{\large\textbullet}}]
      \item Steak (flank preferably) – 1 lb
      \item Baking soda – 1 tsp
      \item Brown sugar – 1/3 cup
      \item Corn starch – 1/3 cup
      \item Corn starch slurry (corn starch and water) – 1 tbsp
      \item Garlic – 1 tsp
      \item Ginger – 1 tsp
      \item Salt – 1 tsp
      \item Soy sauce – 1/4 cup
      \item Thai chili and green beans (optional)
\end{itemize}

\makeRecipeTitle
\section*{Instructions}

\begin{enumerate}[label=\textbf{\arabic*.}, itemsep=0.25em]

      \item \textbf{Prep Beef} \\
            Thinly cut your beef using proper knife safety. Mix it with salt, cornstarch, and baking soda. Set aside.

      \item \textbf{Prepare Sauce} \\
            Combine brown sugar, light soy sauce, and broth powder mixture in a bowl or container.

      \item \textbf{Cook Beef} \\
            Fry beef in a pan until golden brown and cooked through. Set aside.

      \item \textbf{Make Sauce} \\
            Add the sauce mixture and the corn starch slurry to the pan.

      \item \textbf{Thicken} \\
            Stir the sauce until it thickens to your desired consistency.

      \item \textbf{Combine} \\
            Add the cooked beef, chili, and green beans (if using) back to the sauce and mix until well coated.

\end{enumerate}
