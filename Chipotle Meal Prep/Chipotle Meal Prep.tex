\section*{Ingredients}

\textbf{Pickled Red Onion}
\begin{itemize}
    \item Red onion – ½, thinly sliced
    \item Water – 1 cup (240ml)
    \item White distilled vinegar – 1 cup (240ml)
    \item Kosher salt – 1 tbsp (10g)
\end{itemize}

\textbf{Rice}
\begin{itemize}
    \item Long grain white rice – 3 cups (600g)
    \item Water – 3 cups (709ml)
    \item Cilantro – 2.5 tbsp (6g), finely chopped
    \item Unsalted butter – 2 tbsp (28g) (optional)
    \item Lime – Juice and zest of 1
    \item Salt – To taste
\end{itemize}

\textbf{Chicken}
\begin{itemize}
    \item Chipotle peppers in adobo – 1 can (7-ounce)
    \item Kosher salt – 2.5 tsp (8g)
    \item Garlic – 4 cloves, grated
    \item Boneless, skinless chicken thighs – 6 (about 1.5 lbs / 680g)
    \item Vegetable oil or cooking spray
\end{itemize}

\section*{Instructions}

\begin{enumerate}

    \item \textbf{For the Pickled Red Onion} \\
          Add sliced onion to a 1-quart heatproof jar or container. In a medium saucepan over medium-high heat, combine water, vinegar, and salt. When the mixture begins to boil, remove from heat and pour over the red onions to cover. Let the mixture come to room temperature, uncovered. Once cooled, cover with a tight-fitting lid and refrigerate. (Stored for a few months).

    \item \textbf{For the Rice} \\
          Rinse rice thoroughly under cold water until it runs clear. Strain dry and cook in rice cooker with 3 cups water. Once cooked, for added flavor, stir in cilantro, butter, lime zest, and juice until evenly combined. Season with salt to taste and keep warm.

    \item \textbf{For the Chicken} \\
          Strain liquid from chipotle can. Roughly chop peppers and add back to liquid with salt and garlic. Pour over chicken thighs in a bowl; coat evenly. Marinate 30 mins to overnight. \\
          Heat a cast iron/grill pan over medium-high heat with oil. Sear chicken in batches (don't overcrowd) for 2–3 mins per side until charred and golden. Cook to 165$^\circ$F. If charring too fast, finish in oven at 375$^\circ$F. Rest 5 mins before dicing.

    \item \textbf{For the Black Beans} \\
          In a saucepan over medium heat with oil, sauté onion, serrano, and tomato until softened (2–3 mins). Stir in black beans, stock, and salt. Boil then reduce to low; simmer until veggies soften and stock reduces by a quarter. Remove from heat and stir in garlic.

    \item \textbf{For the Corn Relish} \\
          Char corn cobs over open flame/grill (1–2 mins/side). Char jalapeños until black; remove seeds/skin if desired and dice. Cut corn off cob. Mix corn, jalapeño, red onion, cilantro. Season with salt/pepper. Stir in vinegar (adjust to taste).

    \item \textbf{For the Spicy Mayo} \\
          Whisk all ingredients in a bowl. Season with salt. Refrigerate.

    \item \textbf{To Assemble} \\
          Divide rice between bowls (~1 cup each). Top with diced chicken, corn relish (with spicy mayo), black beans, pickled red onions, cheese, and cilantro. Enjoy!

\end{enumerate}
