\makeRecipeTitle{Butter Chicken}

\section*{Ingredients}

\subsection*{For the Chicken Marinade:}

\begin{itemize}
      \item Boneless and Skinless Chicken Thighs or Breasts – 28 oz (cut into bite-sized pieces)
      \item Plain Yogurt – ½ cup
      \item Garlic Minced – 1½ tablespoons
      \item Ginger Minced – 1 tablespoon
      \item Garam Masala – 2 teaspoons
      \item Turmeric – 1 teaspoon
      \item Ground Cumin – 1 teaspoon
      \item Red Chili Powder – 1 teaspoon (Kashmiri recommended)
      \item Salt – 1 teaspoon
\end{itemize}

\subsection*{For the Sauce:}

\begin{itemize}
      \item Olive Oil – 2 tablespoons
      \item Ghee or Butter – 2 tablespoons ghee or 1 tablespoon butter + 1 tablespoon oil
      \item Large Onion – 1 (sliced or chopped)
      \item Garlic Minced – 1½ tablespoons
      \item Ginger Minced – 1 tablespoon
      \item Ground Cumin – 1½ teaspoons
      \item Garam Masala – 1½ teaspoons
      \item Ground Coriander – 1 teaspoon
      \item Crushed Tomatoes – 14 oz
      \item Red Chili Powder – 1 teaspoon (adjust to taste, use Kashmiri)
      \item Salt – 1¼ teaspoons (or to taste)
      \item Thickened Cream – 1 cup (heavy cream or evaporated milk to save calories)
      \item Sugar – 1 tablespoon
      %\item Kasoori Methi – ½ teaspoon (dried fenugreek leaves)
\end{itemize}

\newpage
\makeRecipeTitle{Butter Chicken}
\section*{Instructions}

\begin{enumerate}

      \item \textbf{Marinate Chicken} \\
            In a bowl, mix plain yogurt, garlic, ginger, garam masala, turmeric, cumin, chili powder, and salt. Add chicken pieces and mix so it is coated evenly. Allow to marinate for at least 30 minutes (or overnight in the fridge for deeper flavor).

      \item \textbf{Sear Chicken} \\
            Heat a large skillet or pan over medium-high heat and add oil. Cook the marinated chicken in batches, searing until browned on all sides. Set aside.

      \item \textbf{Sauté Onions and Aromatics} \\
            In the same skillet, melt butter or ghee and sauté onions until softened. Add garlic, ginger, and remaining spices (garam masala, ground coriander, and cumin) and cook until fragrant.

      \item \textbf{Build the Sauce} \\
            Stir in crushed tomatoes, chili powder, and salt. Let the tomatoes cook for about 10--15 minutes until they are nearly dry in your pan, stirring occasionally until the sauce thickens and becomes a deep brown red color.

      \item \textbf{Add Cream and Season} \\
            Add cream to the sauce in the skillet and stir well. Add sugar and adjust seasoning with salt to balance flavors.

      \item \textbf{Simmer with Chicken} \\
            Add the seared chicken with juices back into the sauce and simmer for an additional 8--10 minutes until chicken is cooked through and the sauce is thick and bubbling, allowing the flavors to meld.

      \item \textbf{Serve} \\
            Garnish with fresh chopped cilantro and serve with rice or garlic naan for a complete meal!

\end{enumerate}
