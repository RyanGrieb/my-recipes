\makeRecipeTitle{Creamy Tuscan Chicken}

\section*{Ingredients}

\begin{itemize}
      \item Boneless Skinless Chicken Breasts – 8 (6-ounce)
      \item Italian Seasoning – 2 tsp
      \item Kosher Salt – 2 tsp
      \item Black Pepper – 1 tsp
      \item Extra Virgin Olive Oil – 6 tbsp
      \item Garlic Cloves – 8, minced
      \item Yellow Onion – ½ cup, finely chopped
      \item Sun-dried Tomatoes – 1 cup, thinly sliced
      \item Baby Spinach – 4 cups
      \item Heavy Cream – 3 cups
      \item Parmesan Cheese – ½ cup, freshly grated
\end{itemize}

\newpage
\makeRecipeTitle{Creamy Tuscan Chicken}
\section*{Instructions}

\begin{enumerate}

      \item \textbf{Season} \\
            Season chicken with Italian seasoning, salt, and black pepper.

      \item \textbf{Sear Chicken} \\
            Heat 2 tbsp of olive oil in a large skillet over medium heat. Add the chicken and sear for 3–4 minutes on each side until golden. Remove chicken and set aside.

      \item \textbf{Sauté Aromatics} \\
            Add 1 tbsp of oil to the same skillet and add the onion. Sauté for 2–3 minutes until soft. Add the tomatoes and garlic, sauté another minute until fragrant.

      \item \textbf{Wilt Spinach} \\
            Add the spinach, sauté another minute until just starting to wilt.

      \item \textbf{Make Sauce} \\
            Add heavy cream and parmesan cheese. Stir together and bring to a simmer.

      \item \textbf{Finish Cooking} \\
            Put chicken back into the skillet and cook until cooked through (165$^\circ$F), about 5 minutes.

\end{enumerate}
